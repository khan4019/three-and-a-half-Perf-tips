local variable

store out of scope variable


var myArr = [0, 1, 2,3 4, 5 ...];

function useGlobal() {
    
    for (j = 0; j < 1000; j++){
        console.log(myArr[j]);
    }
}
							

function useLocal() {
    var localArr = myArr;

    for (var k = 0; k < 1000; k++){
        console.log(localArr[k]);
    }
}
